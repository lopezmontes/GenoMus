norm2hertz:
p = \sqrt[4]{\frac{Hz}{20000}}
Hz = 20000p^4

norm2notevalue:
p = 2^{(10v - 8)}
v = \frac{\log(p) + 8\log(2)}{10\log(2)}

norm2seconds:
p = 2^{(10v - 6)}
s = \frac{\log(p) + 6\log(2)}{10\log(2)}

norm2midipitch:
p = \frac{m - 12}{100}
m = 100p+12

norm2articulation:
p = 3a^e
a = \Big(\frac{p}{3}\Big)^\frac{1}{e}

norm2intensity
p = \frac{i}{127}
i = 127p
